module.exports = async function (context, req) {
  context.log("üì© Contact form triggered!");

  try {
    // Ensure req.body is parsed safely
    let data = {};
    if (req.body) {
      data = req.body;
    } else if (req.rawBody) {
      try {
        data = JSON.parse(req.rawBody);
      } catch {
        context.log("‚ö†Ô∏è rawBody was not valid JSON");
      }
    }

    const name = data.name || "";
    const email = data.email || "";
    const message = data.message || "";

    if (!name || !email || !message) {
      context.res = {
        status: 400,
        headers: { "Content-Type": "application/json" },
        body: { success: false, error: "All fields are required." }
      };
      return;
    }

    context.log(`‚úÖ Received message from ${name} <${email}>: ${message}`);

    context.res = {
      status: 200,
      headers: { "Content-Type": "application/json" },
      body: { success: true, message: "Message received successfully." }
    };
  } catch (err) {
    context.log("‚ùå Error:", err.message);
    context.res = {
      status: 500,
      headers: { "Content-Type": "application/json" },
      body: { success: false, error: err.message }
    };
  }
};
